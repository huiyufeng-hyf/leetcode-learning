```
public class Solution {
    public string RemoveDuplicates(string s) {
        var sb = new StringBuilder();
        var top = -1;
        for (var i = 0; i < s.Length; i++)
        {
            var ch = s[i];
            if (top >= 0 && sb[top] == ch)
            {
                sb.Remove(top, 1);
                --top;
            }
            else
            {
                sb.Append(ch);
                ++top;
            }
        }

        return sb.ToString();
    }
}
```

```
class Solution:
    def removeDuplicates(self, s: str) -> str:
        stk = list()
        for ch in s:
            if stk and stk[-1] == ch:
                stk.pop()
            else:
                stk.append(ch)
        return "".join(stk)
```
