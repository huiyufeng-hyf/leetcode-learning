- Solution by enumeration
```
public class Solution {
    public string SortSentence(string s) {
        var dict = new Dictionary<int, string>();
        var wordArr = s.Split(' ', StringSplitOptions.RemoveEmptyEntries);
        foreach (var word in wordArr)
        {
            var len = word.Length;
            dict.Add(Convert.ToInt32(word[len - 1]), word.Substring(0, len - 1));
        }
        var sortedList = dict.OrderBy(i => i.Key).Select(i => i.Value).ToList();
        var sb = new StringBuilder();
        foreach (var item in sortedList)
        {
            sb.Append(item);
            sb.Append(" ");
        }
        return sb.ToString().Trim();
    }
}

**Python**
```
class Solution:
    def sortSentence(self, s: str) -> str:
        word_dict = {}
        word_arr = s.split()

        for word in word_arr:
            index = int(word[-1])
            word_dict[index] = word[:-1]

        sorted_words = [word_dict[key] for key in sorted(word_dict.keys())]

        return ' '.join(sorted_words)

class Solution:
    def sortSentence(self, s: str) -> str:
        words = s.split()
        return ' '.join(word[:-1] for word in sorted(words, key=lambda w: int(w[-1])))

- Recommendation by maintaining the word array
```
class Solution:
    def sortSentence(self, s: str) -> str:
        words = s.split()
        n = len(s)
        arr = ["" for _ in range(n)]
        for wd in words:
            arr[int(wd[-1]) - 1] = wd[:-1]
        return " ".join(arr).strip()
        
```
