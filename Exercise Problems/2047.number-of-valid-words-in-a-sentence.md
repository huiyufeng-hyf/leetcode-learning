public class Solution {
    public int CountValidWords(string sentence) {
        var tokenArr = sentence.Split(' ', StringSplitOptions.RemoveEmptyEntries);
        var cnt = 0;
        foreach (var token in tokenArr)
        {
            var hasDigit = false;
            var hyphenCnt = 0;
            var isValidHypen = true;
            var puncCnt = 0;
            var isValidPunc = true;
            var n = token.Length;
            for (var i = 0; i < n; i++)
            {
                if (char.IsDigit(token[i]))
                {
                    hasDigit = true;
                    break;
                }
                if (token[i] == '-')
                {
                    hyphenCnt++;
                    if (i == 0 || i == n - 1)
                    {
                        isValidHypen = false;
                        break;
                    }
                    else if (!char.IsLetter(token[i - 1]) || !char.IsLetter(token[i + 1]))
                    {
                        isValidHypen = false;
                        break;
                    }
                }
                if (token[i] == '!' || token[i] == '.' || token[i] == ',')
                {
                    puncCnt++;
                    if (i != n - 1)
                    {
                        isValidPunc = false;
                        break;
                    }
                }
            }
            if (hasDigit || (!isValidHypen || hyphenCnt > 1) || (!isValidPunc || puncCnt > 1))
                continue;

            cnt++;
        }
        return cnt;
    }
}

```
class Solution:
    def countValidWords(self, sentence: str) -> int:
        token_arr = sentence.split()
        cnt = 0
        for token in token_arr:
            has_digit = False
            hyphen_cnt, is_valid_hyphen = 0, True
            punc_cnt, is_valid_punc = 0, True
            n = len(token)
            for i in range(n):
                if (token[i].isdigit()):
                    has_digit = True
                    break
                if token[i] == '-':
                    hyphen_cnt += 1
                    if i == 0 or i == n - 1:
                        is_valid_hyphen = False
                        break
                    elif not token[i-1].isalpha() or not token[i + 1].isalpha():
                        is_valid_hyphen = False
                        break
                if token[i] == '!' or token[i] == '.' or token[i] == ',':
                    punc_cnt += 1
                    if i != n - 1:
                        is_valid_punc = False
                        break
            if has_digit or not is_valid_hyphen or hyphen_cnt > 1 or not is_valid_punc or punc_cnt > 1:
                continue
            
            cnt += 1
        return cnt

```
class Solution:
    def countValidWords(self, sentence: str) -> int:
        def valid(s: str) -> bool:
            has_hyphens = False
            for i, ch in enumerate(s):
                if ch.isdigit() or ch in "!.," and i < len(s) - 1:
                    return False
                if ch == '-':
                    if has_hyphens or i == 0 or i == len(s) - 1 or not s[i - 1].islower() or not s[i + 1].islower():
                        return False
                    has_hyphens = True
            return True
        
        return sum(valid(s) for s in sentence.split())
