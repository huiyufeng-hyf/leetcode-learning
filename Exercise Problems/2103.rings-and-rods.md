- Solution by hashset
```
public class Solution {
    public int CountPoints(string rings) {
        var n = rings.Length;
        var rSet = new HashSet<int>();
        var gSet = new HashSet<int>();
        var bSet = new HashSet<int>();
        for (var i = 0; i < n - 1; i += 2)
        {
            var rodNum = (int)rings[i + 1];
            if (rings[i] == 'R')
                rSet.Add(rodNum);
            else if (rings[i] == 'G')
                gSet.Add(rodNum);
            else
                bSet.Add(rodNum);
        }

        var finalSet = rSet.Intersect(gSet).Intersect(bSet);
        return finalSet.Count();
    }
}

```
class Solution:
    def countPoints(self, rings: str) -> int:
        cnt = 0
        n = len(rings)
        rset, gset, bset = set(), set(), set()
        for i in range(0, n, 2):
            rod_num = int(rings[i + 1])
            if rings[i] == 'R':
                rset.add(rod_num)
            elif rings[i] == 'G':
                gset.add(rod_num)
            else:
                bset.add(rod_num)
        for num in rset:
            if num in gset and num in bset:
                cnt += 1
        return cnt
```

- Recommendation 1 by maintaining the state of each pole
```
class Solution:
    def countPoints(self, rings: str) -> int:
        def getColorId(color: char) -> bool:
            if color == 'R':
                return 0
            elif color == 'G':
                return 1
            else:
                return 2
        
        POLE_NUM, COLOR_NUM = 10, 3
        state = [[0 for i in range(COLOR_NUM)] for j in range(POLE_NUM)]
        n = len(rings)
        for i in range(0, n, 2):
            color = rings[i]
            pole_index = ord(rings[i + 1]) - ord('0')
            state[pole_index][getColorId(color)] = 1
        res = 0
        for i in range(POLE_NUM):
            flag = True
            for j in range(COLOR_NUM):
                if state[i][j] == 0:
                    flag = False
                    break
            if flag:
                res += 1
        return res
```

- Recommendation 2 by state optimization
```
class Solution:
    def countPoints(self, rings: str) -> int:
        POLE_NUM = 10
        mapping = {'R': 0, 'G': 1, 'B': 2}
        state = [0] * POLE_NUM
        for i in range(0, len(rings), 2):
            color = rings[i]
            pole_index = ord(rings[i + 1]) - ord('0')
            state[pole_index] |= 1 << mapping[color]
        return sum(state[i] == 7 for i in range(POLE_NUM))
