```
public class Solution {
    public int CountVowelSubstrings(string word) {
        var cnt = 0;
        var n = word.Length;
        var vowelSet = new HashSet<char> {'a', 'e', 'i', 'o', 'u'};
        for (var i = 0; i < n; i++)
        {
            if (!vowelSet.Contains(word[i]))
                continue;
            
            var seen = new HashSet<char>();
            for (var j = i; j < n; j++)
            {
                if (!vowelSet.Contains(word[j]))
                    break;
                seen.Add(word[j]);
                if (seen.Count == 5)
                    cnt++;
            }
        }
        return cnt;
    }
}

**Python**
```
class Solution:
    def countVowelSubstrings(self, word: str) -> int:
        cnt, n = 0, len(word)
        vowel_set = set({'a', 'e', 'i', 'o', 'u'})
        for i in range(n):
            if not word[i] in vowel_set:
                continue
            
            seen = set()
            for j in range(i, n):
                if not word[j] in vowel_set:
                    break
                seen.add(word[j])
                if len(seen) == 5:
                    cnt += 1
        return cnt
        
```

```
class Solution:
    def countVowelSubstrings(self, word: str) -> int:
        n = len(word)
        res = 0
        vowelset = set("aeiou")
        for i in range(n):
            charset = set()
            for j in range(i, n):
                charset.add(word[j])
                if charset == vowelset:
                    res += 1
        return res
