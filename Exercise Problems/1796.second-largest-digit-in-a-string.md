- Solution 1 by checking digit
```
public class Solution {
    public int SecondHighest(string s) {
        if (s.Length <= 1) return -1;

        var list = new List<int>();
        foreach (var ch in s)
        {
            if (char.IsDigit(ch))
            {
                var d = ch - '0';
                if (!list.Contains(d))
                    list.Add(d);
            }
        }

        if (list.Count <= 1) return -1;
        list.Sort();
        return list[list.Count - 2];
    }
}
```
**Python**
```
class Solution:
    def secondHighest(self, s: str) -> int:
        if len(s) <= 1:
            return -1
        
        digits = []
        for ch in s:
            if ch.isdigit():
                d = int(ch)
                if d not in digits:
                    digits.append(d)
        
        if len(digits) <= 1:
            return -1
        
        digits.sort()
        return digits[-2]
```

- Recommendation by direct enumerating
```
class Solution:
    def secondHighest(self, s: str) -> int:
        first = second = -1
        for c in s:
            if c.isdigit():
                num = int(c)
                if num > first:
                    second = first
                    first = num
                elif second < num < first:
                    second = num
        return second
```
