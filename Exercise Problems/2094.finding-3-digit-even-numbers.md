```
public class Solution {
    public int[] FindEvenNumbers(int[] digits) {
        var n = digits.Length;
        var res = new SortedSet<int>();
        for (var i = 0; i < n; i++)
        {
            for (var j = 0; j < n; j++)
            {
                if (j == i) continue;
                for (var k = 0; k < n; k++)
                {
                    if (k == i || k == j) continue;
                    if (digits[i] != 0 && digits[k] % 2 == 0)
                        res.Add(digits[i] * 100 + digits[j] * 10 + digits[k]);
                }
            }
        }
        return res.ToArray();
    }
}
```

```
class Solution:
    def findEvenNumbers(self, digits: List[int]) -> List[int]:
        n = len(digits)
        res = set()
        for i in range(n):
            for j in range(n):
                if j == i:
                    continue
                for k in range(n):
                    if k == i or k == j:
                        continue
                    if digits[i] != 0 and digits[k] % 2 == 0:
                        res.add(digits[i] * 100 + digits[j] * 10 + digits[k])
        return sorted(res)
```

- Recommendation 1 by enumerating element combination in the array
```
class Solution:
    def findEvenNumbers(self, digits: List[int]) -> List[int]:
        nums = set()
        n = len(digits)
        for i in range(n):
            for j in range(n):
                for k in range(n):
                    if i == j or j == k or i == k:
                        continue
                    num = digits[i] * 100 + digits[j] * 10 + digits[k]
                    if num >= 100 and num % 2 == 0:
                        nums.add(num)
        res = sorted(nums)
        return res
```

- Recommendation 2 by enumerating all possible even nums
```
class Solution:
    def findEvenNumbers(self, digits: List[int]) -> List[int]:
        res = []
        freq = Counter(digits)
        for i in range(100, 1000, 2):
            freq1 = Counter([int(d) for d in str(i)])
            if all(freq[d] >= freq1[d] for d in freq1.keys()):
                res.append(i)
        return res
```
