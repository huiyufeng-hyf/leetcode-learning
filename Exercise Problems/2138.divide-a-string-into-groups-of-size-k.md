**https://leetcode.cn/problems/divide-a-string-into-groups-of-size-k/description/**

- Solution by simulation
```
public class Solution {
    public string[] DivideString(string s, int k, char fill) {
        var ans = new List<string>();
        var n = s.Length;
        for (var i = 0; i < n; i += k)
        {
            if (i + k <= n)
            {
                var subStr = s.Substring(i, k);
                ans.Add(subStr);
            }   
            else
            {
                var subStr = s.Substring(i);
                ans.Add(subStr + new string(fill, k - subStr.Length));
            }
        }
        return ans.ToArray();
    }
}
```
**Python**
```
class Solution:
    def divideString(self, s: str, k: int, fill: str) -> List[str]:
        ans = []
        n = len(s)
        for i in range(0, n, k):
            if i + k <= n:
                ans.append(s[i:i + k])
            else:
                ans.append(s[i:] + "".join([fill] * (k - len(s[i:]))))
        return ans
```

```
class Solution:
    def divideString(self, s: str, k: int, fill: str) -> List[str]:
        ans = []
        n = len(s)
        curr = 0
        while curr < n:
            ans.append(s[curr:curr+k])
            curr += k
        ans[-1] += fill * (k - len(ans[-1]))
        return ans
```
