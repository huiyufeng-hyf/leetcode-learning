**https://leetcode.cn/problems/check-if-all-characters-have-equal-number-of-occurrences/description/**

- Solution by enumeration
```
public class Solution {
    public bool AreOccurrencesEqual(string s) {
        var arr = new int[26];
        foreach (var ch in s)
        {
            var idx = ch - 'a';
            arr[idx]++;
        }

        var cnt = 0;
        for (var i = 0; i < 26; i++)
        {
            if (arr[i] == 0)
                continue;
            if (cnt != 0 && cnt != arr[i])
                return false;
            else
                cnt = arr[i];
        }
        return true;
    }
}
```
**Python**
```
class Solution:
    def areOccurrencesEqual(self, s: str) -> bool:
        arr = [0] * 26
        cnt = 0
        for ch in s:
            idx = ord(ch) - ord('a')
            arr[idx] += 1
        for i in range(26):
            if arr[i] == 0:
                continue
            if cnt != 0 and cnt != arr[i]:
                return False
            else:
                cnt = arr[i]
        return True
```

- Recommendation by counting the frequency
```
class Solution:
    def areOccurrencesEqual(self, s: str) -> bool:
        freq = Counter(s)
        occ = len(s) // len(freq)
        return all(v == occ for v in freq.values())
```
