**https://leetcode.cn/problems/delete-characters-to-make-fancy-string/description/**

- Solution by enumeration
```
public class Solution {
    public string MakeFancyString(string s) {
        var charArr = s.ToCharArray();
        for (var i = 0; i < s.Length - 2; i++)
        {
            if (charArr[i] == charArr[i + 1] && charArr[i] == charArr[i + 2])
                charArr[i] = ' ';
        }
        
        var sb = new StringBuilder();
        foreach (var ch in charArr)
        {
            if (ch != ' ')
                sb.Append(ch);
        }
        return sb.ToString();
    }
}
```
**Python**
```
class Solution:
    def makeFancyString(self, s: str) -> str:
        n = len(s)
        char_arr = [ch for ch in s]
        for i in range(n - 2):
            if char_arr[i] == char_arr[i + 1] and char_arr[i] == char_arr[i + 2]:
                char_arr[i] = ' '
        return "".join([ch for ch in char_arr if ch != ' '])
        
```

class Solution:
    def makeFancyString(self, s: str) -> str:
        res = []
        for ch in s:
            if len(res) >= 2 and res[-1] == res[-2] == ch:
                continue
            res.append(ch)
        return "".join(res)
