```
public class Solution {
    public string ReplaceDigits(string s) {
        var arr = s.ToCharArray();
        for (var i = 1; i < s.Length; i += 2)
        {
            arr[i] = Shift(s[i - 1], Convert.ToInt32(s[i].ToString()));
        }
        return new string(arr);
    }

    private static char Shift(char ch, int d)
    {
        if (d == 0)
            return ch;
        return (char)(ch + d);
    }
}
```

**Python**
```
class Solution:
    def replaceDigits(self, s: str) -> str:
        arr = list(s)
        for i in range(1, len(s)):
            if i % 2 == 1:
                arr[i] = self._shift(s[i - 1], int(s[i]))
        return ''.join(arr)

    def _shift(self, ch: str, d: int) -> str:
        if d == 0:
            return ch
        return chr(ord(ch) + d)
```
