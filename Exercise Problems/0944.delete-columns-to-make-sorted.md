public class Solution {
    public int MinDeletionSize(string[] strs) {
        var len = strs[0].Length;
        var n = strs.Length;
        var deleteCount = 0;
        for (var i = 0; i < len; i++)
        {
            for (var j = 1; j < n; j++)
            {
                if (strs[j][i] - strs[j - 1][i] < 0)
                {
                    deleteCount++;
                    break;
                }
            }
        }

        return deleteCount;
    }
}
