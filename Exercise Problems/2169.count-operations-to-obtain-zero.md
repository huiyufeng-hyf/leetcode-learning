**https://leetcode.cn/problems/count-operations-to-obtain-zero/description/**

- Solution by simulation
```
public class Solution {
    public int CountOperations(int num1, int num2) {
        if (num1 == 0 || num2 == 0)
            return 0;
            
        var ans = 0;
        while (true)
        {
            if (num1 >= num2)
                num1 -= num2;
            else
                num2 -= num1;
            ans += 1;
            if (num1 == 0 || num2 == 0)
                break;
        }
        return ans;
    }
}
```

class Solution:
    def countOperations(self, num1: int, num2: int) -> int:
        if num1 == 0 or num2 == 0:
            return 0
        ans = 0
        while True:
            if num1 >= num2:
                num1 -= num2
            else:
                num2 -= num1
            ans += 1
            if num1 == 0 or num2 == 0:

- Recommendation by sequence dividing
```
class Solution:
    def countOperations(self, num1: int, num2: int) -> int:
        res = 0
        while num1 and num2:
            res += num1 // num2
            num1 %= num2
            num1, num2 = num2, num1
        return res
```
