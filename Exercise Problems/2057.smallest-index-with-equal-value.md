**https://leetcode.cn/problems/smallest-index-with-equal-value/description/**

- Solution by enumeration
```
public class Solution {
    public int SmallestEqual(int[] nums) {
        var minIdx = 100;
        var flag = false;
        for (var i = 0; i < nums.Length; i++)
        {
            if (i % 10 == nums[i])
            {
                flag = true;
                minIdx = Math.Min(i, minIdx);
            }
        }
        if (flag)
            return minIdx;
        return -1;
    }
}
```
**Python**
```
class Solution:
    def smallestEqual(self, nums: List[int]) -> int:
        min_idx, flag = 100, False
        for i in range(len(nums)):
            if i % 10 == nums[i]:
                flag = True
                min_idx = min(i, min_idx)
        if flag:
            return min_idx
        return -1
        
```

- Recommendation by enumeration
```
class Solution:
    def smallestEqual(self, nums: List[int]) -> int:
        n = len(nums)
        for i in range(n):
            if i % 10 == nums[i]:
                return i
        return -1
        
```
