public class Solution {
    public int SubsetXORSum(int[] nums) {
        var n = nums.Length;
        if (n == 0)
            return 0;

        var sum = 0;
        for (var i = 0; i < (1 << n); i++)
        {
            var xor = 0;
            for (var j = 0; j < n; j++)
            {
                if ((i & (1 << j)) != 0)
                    xor ^= nums[j];
            }
            sum += xor;
        }
        
        return sum;
    }
}

class Solution:
    def subsetXORSum(self, nums: List[int]) -> int:
        n = len(nums)
        if n == 0:
            return 0
        
        total_sum = 0

        for i in range(1, 1 << n):
            xor = 0
            for j in range(n):
                if i & (1 << j):
                    xor ^= nums[j]
            total_sum += xor
        
        return total_sum

class Solution:
    def subsetXORSum(self, nums: List[int]) -> int:
        res = 0
        n = len(nums)
        def dfs(val, idx):
            nonlocal res
            if idx == n:
                res += val
                return
            dfs(val ^ nums[idx], idx + 1)
            dfs(val, idx + 1)
        
        dfs(0, 0)
        return res
