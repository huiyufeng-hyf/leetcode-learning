**https://leetcode.cn/problems/count-integers-with-even-digit-sum/description/**

- Solution by enumeration
```
public class Solution {
    public int CountEven(int num) {
        var ans = 0;
        while (num > 0)
        {
            var temp = num;
            var sum = 0;
            while (temp > 0)
            {
                var d = temp % 10;
                sum += d;
                temp /= 10;
            }
            if (sum % 2 == 0)
                ans++;
            num--;
        }
        return ans;
    }
}
```
**Pythyon**

```
class Solution:
    def countEven(self, num: int) -> int:
        ans = 0
        while num > 0:
            temp_num, temp_sum = num, 0
            while temp_num > 0:
                d = temp_num % 10
                temp_sum += d
                temp_num //= 10
            if temp_sum % 2 == 0:
                ans += 1
            num -= 1
        return ans
```

- Recommendation 1 by vialent enumeration
```
class Solution:
    def countEven(self, num: int) -> int:
        ans = 0
        for x in range(1, num + 1):
            s = 0
            while x:
                s += x % 10
                x //= 10
            ans += s % 2 == 0
        return ans
```

```
class Solution:
    def countEven(self, num: int) -> int:
        y, x = divmod(num, 10)
        ans = y * 5
        y_sum = 0
        while y:
            y_sum += y % 10
            y //= 10
        return ans + ((x + 1) // 2 - 1 if y_sum % 2 else x // 2)
```
