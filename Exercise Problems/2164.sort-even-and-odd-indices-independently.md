**https://leetcode.cn/problems/sort-even-and-odd-indices-independently/description/**

- Solution by simulation
```
public class Solution {
    public int[] SortEvenOdd(int[] nums) {
        var evenArr = new List<int>();
        var oddArr = new List<int>();
        var n = nums.Length;
        for (var i = 0; i < n; i++)
        {
            if (i % 2 == 0)
                evenArr.Add(nums[i]);
            else
                oddArr.Add(nums[i]);
        }
        evenArr.Sort();
        oddArr = oddArr.OrderByDescending(x => x).ToList();
        var ans = new int[n];
        var idx = 0;
        foreach (var x in evenArr)
        {
            ans[idx] = x;
            idx += 2;
        }
        idx = 1;
        foreach (var y in oddArr)
        {
            ans[idx] = y;
            idx += 2;
        }
        return ans;
    }
}

**Python**
```
class Solution:
    def sortEvenOdd(self, nums: List[int]) -> List[int]:
        even_arr = list()
        odd_arr = list()
        n = len(nums)
        for i in range(n):
            if i % 2 == 0:
                even_arr.append(nums[i])
            else:
                odd_arr.append(nums[i])
        even_arr.sort()
        odd_arr.sort(reverse=True)
        ans = [0] * n
        idx = 0
        for x in even_arr:
            ans[idx] = x
            idx += 2
        idx = 1
        for y in odd_arr:
            ans[idx] = y
            idx += 2
        return ans

class Solution:
    def sortEvenOdd(self, nums: List[int]) -> List[int]:
        even = sorted(nums[::2])
        odd = sorted(nums[1::2])[::-1]
        nums[::2] = even
        nums[1::2] = odd
        return nums
