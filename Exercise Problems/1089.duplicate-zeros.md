- Solution 1:
```
public class Solution {
    public void DuplicateZeros(int[] arr) {
        var n = arr.Length;
        var newArr = new int[n];
        for (int i = 0, j = 0; i < n && j < n; i++, j++)
        {
            if (arr[i] != 0)
                newArr[j] = arr[i];
            else
            {
                newArr[j] = 0;
                newArr[j++] = 0;
            }
        }

        newArr.CopyTo(arr, 0);
    }
}

```
public class Solution {
    public void DuplicateZeros(int[] arr) {
        var n = arr.Length;
        var top = 0;
        var i = -1;
        while (top < n)
        {
            i++;
            if (arr[i] != 0)
                top++;
            else
                top += 2;
        }

        var j = n - 1;
        if (top == n + 1)
        {
            arr[j] = 0;
            j--;
            i--;
        }

        while (j >= 0)
        {
            arr[j] = arr[i];
            j--;
            if (arr[i] == 0)
            {
                arr[j] = arr[i];
                j--;
            }
            i--;
        }
    }
}
