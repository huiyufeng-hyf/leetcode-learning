- Solution 1:
```
public class Solution {
    public int RepeatedNTimes(int[] nums) {
        var n = nums.Length / 2;
        Array.Sort(nums);
        var ans = nums[0];
        var count = 1;
        for (var i = 1; i < 2 * n; i++)
        {
            if (nums[i] == ans)
                count++;
            else
            {
                if (count == n) 
                    return ans;
                else
                {
                    ans = nums[i];
                    count = 1;
                }
            }
        }

        if (count == n) return ans;

        return 0;
    }
}
```
