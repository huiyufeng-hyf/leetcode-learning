- Solution 1:
```
public class Solution {
    public bool ValidMountainArray(int[] arr) {
        var len = arr.Length;
        if (len <= 2) return false;
        if (arr[0] > arr[1]) return false;

        var descFlag = false;
        for (var i = 0; i < len - 1; i++)
        {
            if (!descFlag)
            {
                if (arr[i] < arr[i + 1]) continue;
                if (arr[i] == arr[i + 1]) return false;
            }
            if (arr[i] > arr[i + 1]) 
            {
                descFlag = true;
                continue;
            }
            if (arr[i] == arr[i + 1]) return false;
            if (arr[i] < arr[i + 1]) 
            {
                descFlag = false;
                return false;
            }
        }

        return descFlag;
    }
}

```
public class Solution {
    public bool ValidMountainArray(int[] arr) {
        var len = arr.Length;
        var i = 0;

        while (i + 1 < len && arr[i] < arr[i + 1])
            i++;
        
        if (i == 0 || i == len - 1)
            return false;
        
        while (i + 1 < len && arr[i] > arr[i + 1])
            i++;

        return i == len - 1;
    }
}
