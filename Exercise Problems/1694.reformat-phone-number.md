- Solution by using Queue
```
public class Solution {
    public string ReformatNumber(string number) {
       var queue = new Queue<char>();
       foreach (var ch in number)
       {
            if (ch != ' ' && ch != '-')
                queue.Enqueue(ch);
       }

       var sb = new StringBuilder();
       var cnt = 0;
       while (queue.Count > 0)
       {
            var digit = queue.Dequeue();
            sb.Append(digit);
            cnt++;
            if (cnt % 3 == 0 && queue.Count != 0 && (queue.Count % 3 == 0 || queue.Count % 2 == 0 || queue.Count > 4))
            {
                sb.Append('-');
                cnt = 0;
            }
            else if (cnt == 2 && queue.Count == 0)
            {
                break;
            }
            else if (cnt == 2 && queue.Count == 2)
            {
                sb.Append('-');
                cnt = 0;
            }  
       }

       return sb.ToString();
    }
}

from collections import deque

class Solution:
    def reformatNumber(self, number: str) -> str:
        queue = deque()
        for ch in number:
            if ch != ' ' and ch != '-':
                queue.append(ch)

        list, cnt = [], 0
        while len(queue) > 0:
            digit = queue.popleft()
            list.append(digit)
            cnt += 1
            if cnt % 3 == 0 and len(queue) != 0 and (len(queue) % 3 == 0 or len(queue) % 2 == 0 or len(queue) > 4):
                list.append('-')
                cnt = 0
            elif cnt == 2 and len(queue) == 0:
                break
            elif cnt == 2 and len(queue) == 2:
                list.append('-')
                cnt = 0
        
        return "".join(list)

- Recommendation by separating the block and then join by "-"
```
class Solution:
    def reformatNumber(self, number: str) -> str:
        digits = list()
        for ch in number:
            if ch.isdigit():
                digits.append(ch)

        n, pt = len(digits), 0
        ans = list()

        while n > 0:
            if n > 4:
                ans.append("".join(digits[pt:pt+3]))
                pt += 3
                n -= 3
            else:
                if n == 4:
                    ans.append("".join(digits[pt:pt+2]))
                    ans.append("".join(digits[pt+2:pt+4]))
                else:
                    ans.append("".join(digits[pt:pt+n]))
                break
        
        return "-".join(ans)
