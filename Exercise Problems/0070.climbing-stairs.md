- Solution 1:
```
public class Solution {
    public int ClimbStairs(int n) {
        switch (n)
        {
            case 1:
                return 1;
            case 2:
                return 2;
        }

        var dp = new int[n + 1];
        dp[1] = 1;
        dp[2] = 2;

        for (var i = 3; i <= n; i++)
        {
            dp[i] = dp[i - 1] + dp[i - 2];
        }

        return dp[n];
    }
}
```
