**https://leetcode.cn/problems/reverse-prefix-of-word/description/**

- Solution by simulation
```
public class Solution {
    public string ReversePrefix(string word, char ch) {
        var idx = word.IndexOf(ch);
        if (idx <= 0)
            return word;

        var prefixStr = word.Substring(0, idx + 1);
        var suffixStr = word.Substring(idx + 1);
        return new string(prefixStr.Reverse().ToArray()) + suffixStr;
    }
}
```
**Python**
```
class Solution:
    def reversePrefix(self, word: str, ch: str) -> str:
        idx = word.index(ch) if ch in word else -1
        if idx <= 0:
            return word
        
        prefix_str = word[0 : idx + 1]
        suffix_str = word[idx + 1:]
        return prefix_str[::-1] + suffix_str
```

class Solution:
    def reversePrefix(self, word: str, ch: str) -> str:
        idx = word.find(ch) + 1
        return word[:idx][::-1] + word[idx:]
