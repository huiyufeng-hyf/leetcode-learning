public class Solution {
    public bool IsThree(int n) {
        // Check if n has exactly three positive divisors
        // Only perfect squares of primes have exactly three divisors
        if (n <= 1)
            return false;
        var sqrt = (int)Math.Sqrt(n);
        if (sqrt * sqrt != n)
            return false;
        // Check if sqrt is prime
        for (var i = 2; i * i <= sqrt; i++)
        {
            if (sqrt % i == 0)
                return false;
        }

        return sqrt > 1;
    }
}

```
class Solution:
    def isThree(self, n: int) -> bool:
        # Check if n has exactly three positive divisors
        # Only perfect squares of primes have exactly three divisors
        if n <= 1:
            return False
        sqrt_val = int(sqrt(n))
        if sqrt_val * sqrt_val != n:
            return False
        # Check if sqrt_val is prime
        for i in range(2, sqrt_val + 1):
            if i * i <= sqrt_val and sqrt_val % i == 0:
                return False
        return sqrt_val > 1
```
