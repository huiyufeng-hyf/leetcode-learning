**https://leetcode.cn/problems/maximum-number-of-words-found-in-sentences/description/**

- Solution 1 by enumeration
```
public class Solution {
    public int MostWordsFound(string[] sentences) {
        var maxCnt = 0;
        foreach (var s in sentences)
        {
            var arr = s.Split(' ', StringSplitOptions.RemoveEmptyEntries);
            maxCnt = Math.Max(maxCnt, arr.Length);
        }
        return maxCnt;
    }
}
```

class Solution:
    def mostWordsFound(self, sentences: List[str]) -> int:
        max_cnt = 0
        for s in sentences:
            arr = s.split(' ')
            max_cnt = max(max_cnt, len(arr))
        return max_cnt

- Solution 2 by space optimization with O(1)
```
public class Solution {
    public int MostWordsFound(string[] sentences) {
        var maxCnt = 0;
        foreach (var s in sentences)
        {
            var tmpCnt = 0;
            foreach (var ch in s)
            {
                if (ch == ' ')
                    tmpCnt++;
            }
            tmpCnt++;
            maxCnt = Math.Max(maxCnt, tmpCnt);
        }
        return maxCnt;
    }
}
```

- Recommendation by counting space
```
class Solution:
    def mostWordsFound(self, sentences: List[str]) -> int:
        max_cnt = 0
        for s in sentences:
            cnt = s.count(' ') + 1
            max_cnt = max(max_cnt, cnt)
        return max_cnt
```
