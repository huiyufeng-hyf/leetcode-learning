- Solution by checking rule
```
public class Solution {
    public bool SquareIsWhite(string coordinates) {
        var ch = coordinates[0];
        var num = Convert.ToInt32(coordinates[1]);

        if ((ch - 'a') % 2 == 0 && num % 2 == 0)
            return true;
        else if ((ch - 'a') % 2 == 1 && num % 2 == 1)
            return true;

        return false;
    }
}
```
**Python**
```
class Solution:
    def squareIsWhite(self, coordinates: str) -> bool:
        ch, num = coordinates[0], int(coordinates[1])

        if (ord(ch) - ord('a')) % 2 == 0 and num % 2 == 0:
            return True
        elif (ord(ch) - ord('a')) % 2 == 1 and num % 2 == 1:
            return True

        return False
        
```
