- Solution by enumeration
```
public class Solution {
    public bool CheckAlmostEquivalent(string word1, string word2) {
        var word1Arr = ParseWord(word1);
        var word2Arr = ParseWord(word2);
        for (var i = 0; i < 26; i++)
        {
            if (Math.Abs(word1Arr[i] - word2Arr[i]) > 3)
                return false;
        }
        return true;
    }

    private int[] ParseWord(string word)
    {
        var arr = new int[26];
        foreach (var ch in word)
        {
            arr[ch - 'a'] += 1;
        }
        return arr;
    }
}
```

**Python**
```
class Solution:
    def checkAlmostEquivalent(self, word1: str, word2: str) -> bool:
        def parse_word(word: str) -> list[int]:
            arr = [0] * 26
            for ch in word:
                arr[ord(ch) - ord('a')] += 1
            return arr
        
        word1_arr = parse_word(word1)
        word2_arr = parse_word(word2)
        for i in range(26):
            if abs(word1_arr[i] - word2_arr[i]) > 3:
                return False
        return True
```

- Recommendation by hashset
```
class Solution:
    def checkAlmostEquivalent(self, word1: str, word2: str) -> bool:
        freq = defaultdict(int)
        for ch in word1:
            freq[ch] += 1
        for ch in word2:
            freq[ch] -= 1
        return all(abs(x) <= 3 for x in freq.values())
```
