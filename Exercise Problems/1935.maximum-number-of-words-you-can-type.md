**https://leetcode.cn/problems/maximum-number-of-words-you-can-type/description/**

- Solution by enumerating each word
```
public class Solution {
    public int CanBeTypedWords(string text, string brokenLetters) {
        var cnt = 0;
        var wordArr = text.Split(' ', StringSplitOptions.RemoveEmptyEntries);
        foreach (var word in wordArr)
        {
            var flag = true;
            foreach (var ch in word)
            {
                if (brokenLetters.IndexOf(ch) >= 0)
                {
                    flag = false;
                    break;
                }
            }
            if (flag)
                cnt++;
        }
        return cnt;
    }
}
```

**Python**
```
class Solution:
    def canBeTypedWords(self, text: str, brokenLetters: str) -> int:
        cnt = 0
        word_arr = text.split(' ')
        for word in word_arr:
            flag = True
            for ch in word:
                if ch in brokenLetters:
                    flag = False
                    break
            if flag:
                cnt += 1
        return cnt
```

```
class Solution:
    def canBeTypedWords(self, text: str, brokenLetters: str) -> int:
        broken = set(brokenLetters)
        res = 0
        flag = True
        for ch in text:
            if ch == ' ':
                if flag:
                    res += 1
                flag = True
            elif ch in broken:
                flag = False
        if flag:
            res += 1
        return res
```
