- Solution by simulation
```
public class Solution {
    public int ArraySign(int[] nums) {
        var product = (double)nums[0];
        for (var i = 1; i < nums.Length; i++)
        {
            if (nums[i] == 0)
            {
                product = 0;
                break;
            }
            else
                product *= nums[i];
        }

        return SignFunc(product);
    }

    private int SignFunc(double x)
    {
        if (x > 0)
            return 1;
        else if (x < 0)
            return -1;
        else
            return 0;
    }
}

**Python**
```
class Solution:
    def arraySign(self, nums: List[int]) -> int:
        product = float(nums[0])

        for i in range(1, len(nums)):
            if nums[i] == 0:
                product = 0
                break
            else:
                product *= nums[i]
        return self._sign_func(product)

    def _sign_func(self, x : float) -> int:
        if x > 0:
            return 1
        elif x < 0:
            return -1
        else:
            return 0

class Solution:
    def arraySign(self, nums: List[int]) -> int:
        sign = 1
        for num in nums:
            if num == 0:
                return 0
            if num < 0:
                sign = -sign
        return sign
