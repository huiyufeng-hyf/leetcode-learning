**https://leetcode.cn/problems/maximum-difference-between-increasing-elements/description/**

- Solution by enumeration
```
public class Solution {
    public int MaximumDifference(int[] nums) {
        var maxDiff = -1;
        var n = nums.Length;
        for (var i = 0; i < n - 1; i++)
        {
            for (var j = i + 1; j < n; j++)
            {
                if (nums[i] < nums[j])
                {
                    var diff = nums[j] - nums[i];
                    maxDiff = Math.Max(diff, maxDiff);
                }
            }
        }
        return maxDiff;
    }
}
```
**Python**
```
class Solution:
    def maximumDifference(self, nums: List[int]) -> int:
        max_diff, n = -1, len(nums)
        for i in range(n - 1):
            for j in range(i + 1, n):
                if nums[i] < nums[j]:
                    diff = nums[j] - nums[i]
                    max_diff = max(diff, max_diff)
        return max_diff
        
```

- Recommendation by pre-min maintenance
```
class Solution:
    def maximumDifference(self, nums: List[int]) -> int:
        n = len(nums)
        ans, premin = -1, nums[0]
        for i in range(1, n):
            if nums[i] > premin:
                ans = max(ans, nums[i] - premin)
            else:
                premin = nums[i]
        return ans
        
```
