```
public class Solution {
    public string MergeAlternately(string word1, string word2) {
        var m = word1.Length;
        var n = word2.Length;
        var preLength = Math.Min(m, n);
        var sb = new StringBuilder();
        for (var i = 0; i < preLength; i++)
        {
            sb.Append(word1[i]);
            sb.Append(word2[i]);
        }
        if (preLength == m)
            sb.Append(word2.Substring(preLength));
        else
            sb.Append(word1.Substring(preLength));
        
        return sb.ToString();
    }
}
```

**Python**
```
class Solution:
    def mergeAlternately(self, word1: str, word2: str) -> str:
        m, n = len(word1), len(word2)
        pre_length = min(m, n)
        arr = []
        for i in range(pre_length):
            arr.append(word1[i])
            arr.append(word2[i])
        if pre_length == m:
            arr.append(word2[pre_length:])
        else:
            arr.append(word1[pre_length:])
        return "".join(arr)
```

```
class Solution:
    def mergeAlternately(self, word1: str, word2: str) -> str:
        m, n = len(word1), len(word2)
        i = j = 0
        ans = list()
        while i < m or j < n:
            if i < m:
                ans.append(word1[i])
                i += 1
            if j < n:
                ans.append(word2[j])
                j += 1

        return "".join(ans)
```
