public class Solution {
    public bool CheckOnesSegment(string s) {
        if (s.Length == 1) return true;

        var prev = '1';
        var totalCnt = 1;
        var tmpCnt = 0;
        var newStart = false;
        for (var i = 1; i < s.Length; i++)
        {
            if (s[i] == prev)
            {
                if (prev == '1' && newStart)
                    tmpCnt++;
            }
            else
            {
                if (prev == '1')
                {
                    if (tmpCnt > 0) totalCnt++;
                    if (totalCnt > 1) return false;
                    tmpCnt = 0;
                    newStart = false;
                }
                else
                {
                    tmpCnt++;
                    newStart = true;
                }
                prev = s[i];
            }
        }
        if (tmpCnt > 0)
            totalCnt++;

        if (totalCnt == 1)
            return true;

        return false;
    }
}

class Solution:
    def checkOnesSegment(self, s: str) -> bool:
        n = len(s)
        if (n == 1):
            return True;

        prev, total_cnt, tmp_cnt, new_start = '1', 1, 0, False
        for i in range(1, n):
            if s[i] == prev:
                if prev == '1' and new_start:
                    tmp_cnt += 1
            else:
                if prev == '1':
                    if tmp_cnt > 0:
                        total_cnt += 1
                    if total_cnt > 1:
                        return False
                    tmp_cnt = 0;
                    new_start = False
                else:
                    tmp_cnt += 1
                    new_start = True
            prev = s[i]
        if tmp_cnt > 0:
            total_cnt += 1
        
        if total_cnt == 1:
            return True
        
        return False
