- Solution 1:
```
public class Solution {
    public bool DivisorGame(int n) {
        if (n == 1) return false;
        var count = 0;
        var num = n;
        var x = 0;
        while (num > 0)
        {
            x = num - 1;
            if ((num % x) == 0)
            {
                count++;
                if (num == n || num == 2) break;
                num = num - x;
            }
            else
            {
                count++;
                num = x;
            }
        }

        return count % 2 != 0 ? true : false;
    }
}
```

- Solution 3: (Recommendation 2)
```
public class Solution {
    public bool DivisorGame(int n) {
        var f = new bool[n + 5];

        f[1] = false;
        f[2] = true;
        for (var i = 3; i <= n; i++)
        {
            for (var j = 1; j < i; j++)
            {
                if (i % j == 0 && !f[i - j])
                {
                    f[i] = true;
                    break;
                }
            }
        }

        return f[n];
    }
}
```
