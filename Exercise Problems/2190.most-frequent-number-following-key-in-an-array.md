**https://leetcode.cn/problems/most-frequent-number-following-key-in-an-array/description/**

- Solution by enumeration with hashset
```
public class Solution {
    public int MostFrequent(int[] nums, int key) {
        var n = nums.Length;
        var dict = new Dictionary<int, int>();
        for (var i = 0; i <= n - 2; i++)
        {
            if (nums[i] == key)
            {
                CountNum(nums[i + 1], dict);
            }
        }
        var maxVal = dict.Max(x => x.Value);
        return dict.Where(x => x.Value == maxVal).First().Key;
    }

    private void CountNum(int num, Dictionary<int, int> dict)
    {
        if (dict.ContainsKey(num))
            dict[num]++;
        else
            dict[num] = 1;
    }
}
```
**Python**
```
class Solution:
    def mostFrequent(self, nums: List[int], key: int) -> int:
        n = len(nums)
        arr = []
        for i in range(n - 1):
            if nums[i] == key:
                arr.append(nums[i + 1])
        res = Counter(arr)
        ans = 0
        cnt = 0
        for k, v in res.items():
            if v > cnt:
                ans = k
                cnt = v
        return ans
```

- Recommendation by hashset
```
class Solution:
    def mostFrequent(self, nums: List[int], key: int) -> int:
        n = len(nums)
        freq = Counter()
        for i in range(n - 1):
            if nums[i] == key:
                freq[nums[i + 1]] += 1
        return freq.most_common(1)[0][0]
```
