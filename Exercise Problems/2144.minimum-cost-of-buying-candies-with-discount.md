**https://leetcode.cn/problems/minimum-cost-of-buying-candies-with-discount/description/**

- Solution by sorting
```
public class Solution {
    public int MinimumCost(int[] cost) {
        var n = cost.Length;
        if (n <= 2)
            return cost.Sum();

        var ans = 0;
        Array.Sort(cost);
        for (var i = n - 1; i >= 0; i -= 3)
        {
            if (i == 0)
            {
                ans += cost[i];
                break;
            }
            ans += cost[i];
            ans += cost[i - 1];
        }
        return ans;
    }
}
```

```
class Solution:
    def minimumCost(self, cost: List[int]) -> int:
        n = len(cost)
        if n <= 2:
            return sum(cost)
        
        ans = 0
        cost.sort()
        for i in range(n - 1, -1, -3):
            if i == 0:
                ans += cost[i]
                break
            ans += cost[i]
            ans += cost[i - 1]
        return ans
```

- Recommendation by greedy with sorting
```
class Solution:
    def minimumCost(self, cost: List[int]) -> int:
        cost.sort(key = lambda x: -x)
        res = 0
        n = len(cost)
        for i in range(n):
            if i % 3 != 2:
                res += cost[i]
        return res
```
