- Solution 1:
```
public class Solution {
    public bool CanConstruct(string ransomNote, string magazine) {
        for(var i=0; i<ransomNote.Length; i++)
        {
            var index = magazine.IndexOf(ransomNote[i]);
            if(index < 0)
                return false;
            else
                magazine = magazine.Remove(index, 1);
        }
        return true;
    }
}
```
