- Solution by enumeration
```
public class Solution {
    public bool CheckIfPangram(string sentence) {
        if (sentence.Length < 26) return false;

        var set = new HashSet<char>();
        foreach (var ch in sentence)
        {
            if (ch >= 97 && ch <= 122)
                set.Add(ch);

            if (set.Count == 26) return true;
        }

        if (set.Count == 26) return true;
        return false;
    }
}
```
**Python**
```
class Solution:
    def checkIfPangram(self, sentence: str) -> bool:
        if len(sentence) < 26:
            return False
        
        chars = set()
        for ch in sentence:
            if ord(ch) >= 97 and ord(ch) <= 122:
                chars.add(ch)
            if len(chars) == 26:
                return True
        
        if len(chars) == 26:
            return True
        return False
        
```

- Recommendation 1 by Array
```
class Solution:
    def checkIfPangram(self, sentence: str) -> bool:
        exist = [False] * 26
        for c in sentence:
            exist[ord(c) - ord('a')] = True
        return all(exist)
        
```

- Recommendation 2 by Binary Collection
```
class Solution:
    def checkIfPangram(self, sentence: str) -> bool:
        state = 0
        for c in sentence:
            state |= 1 << (ord(c) - ord('a'))
        return state == (1 << 26) - 1
        
```
