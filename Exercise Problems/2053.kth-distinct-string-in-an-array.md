**https://leetcode.cn/problems/kth-distinct-string-in-an-array/description/**

- Solution by simulation
```
public class Solution {
    public string KthDistinct(string[] arr, int k) {
        var dict = new Dictionary<string, int>();
        foreach (var s in arr)
        {
            if (dict.ContainsKey(s))
                dict[s]++;
            else
                dict[s] = 1;
        }
        
        var cnt = 0;
        var ans = string.Empty;
        foreach (var (key, val) in dict)
        {
            if (val == 1)
            {
                cnt++;
                if (cnt == k)
                {
                    ans = key;
                    break;
                }
            }
        }
        return ans;
    }
}
```
**Python**
```
class Solution:
    def kthDistinct(self, arr: List[str], k: int) -> str:
        dict_list = dict()
        for s in arr:
            if s in dict_list:
                dict_list[s] += 1
            else:
                dict_list[s] = 1
        cnt, ans = 0, ""
        for key, val in dict_list.items():
            if val == 1:
                cnt += 1
                if cnt == k:
                    ans = key
                    break
        return ans
        
```

- Recommendation by hashset
```
class Solution:
    def kthDistinct(self, arr: List[str], k: int) -> str:
        freq = Counter(arr)
        for s in arr:
            if freq[s] == 1:
                k -= 1
                if k == 0:
                    return s
        return ""
        
```
