- Solution 1:
```
public class Solution {
    public string GenerateTheString(int n) {
        var sb = new StringBuilder();
        if (n % 2 == 0)
        {
            sb.Append("a");
            sb.Append(new string('b', n - 1));
        }
        else
            sb.Append(new string('a', n));
        
        return sb.ToString();
    }
}v
```
