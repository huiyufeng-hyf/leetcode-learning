- Solution by simulation
```
public class Solution {
    public bool HalvesAreAlike(string s) {
        var charArr = new char[]{'a','e','i','o','u','A','E','I','O','U'};
        var n = s.Length;
        var leftCnt = 0;
        var rightCnt = 0;
        for (var i = 0; i < n; i++)
        {
            if (charArr.Contains(s[i]))
            {
                if (i < n / 2)
                    leftCnt++;
                else
                    rightCnt++;
            }
        }

        if (leftCnt == rightCnt) return true;
        return false;
    }
}
```
**Python**
```
class Solution:
    def halvesAreAlike(self, s: str) -> bool:
        chars = ['a','e','i','o','u','A','E','I','O','U']
        n, left_cnt, right_cnt = len(s), 0, 0
        for i, ch in enumerate(s):
            if ch in chars:
                if i < n // 2:
                    left_cnt += 1
                else:
                    right_cnt += 1
        if left_cnt == right_cnt:
            return True
        return False
        
```

- Recommendation by counting
```
class Solution:
    def halvesAreAlike(self, s: str) -> bool:
        vowels = "aeiouAEIOU"
        a, b = s[:len(s) // 2], s[len(s) // 2:]
        return sum(c in vowels for c in a) == sum(c in vowels for c in b)
        
```
