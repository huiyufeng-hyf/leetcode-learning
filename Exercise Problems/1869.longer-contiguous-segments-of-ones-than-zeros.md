- Solution by greedy
```
public class Solution {
    public bool CheckZeroOnes(string s) {
        var maxOne = 0;
        var maxZero = 0;
        var tempCount = 0;
        var curChar = s[0];
        foreach (var ch in s)
        {
            if (ch == curChar)
                tempCount++;
            else
            {
                if (curChar == '1')
                    maxOne = Math.Max(tempCount, maxOne);
                else
                    maxZero = Math.Max(tempCount, maxZero);
                
                curChar = ch;
                tempCount = 1;
            }
        }
        if (curChar == '1')
            maxOne = Math.Max(tempCount, maxOne);
        else
            maxZero = Math.Max(tempCount, maxZero);

        return maxOne > maxZero
    }
}
```

```
class Solution:
    def checkZeroOnes(self, s: str) -> bool:
        max_one, max_zero, temp_count, cur_char = 0, 0, 0, s[0]
        for ch in s:
            if ch == cur_char:
                temp_count += 1
            else:
                if cur_char == '1':
                    max_one = max(temp_count, max_one)
                else:
                    max_zero = max(temp_count, max_zero)
                cur_char = ch
                temp_count = 1
        if cur_char == '1':
            max_one = max(temp_count, max_one)
        else:
            max_zero = max(temp_count, max_zero)
        
        return max_one > max_zero
```
