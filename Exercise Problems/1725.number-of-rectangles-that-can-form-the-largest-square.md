- Solution by simulation:
```
public class Solution {
    public int CountGoodRectangles(int[][] rectangles) {
        var n = rectangles.Length;
        var arr = new int[n];
        var maxLen = 0;
        for (var i = 0; i < rectangles.Length; i++)
        {
            var newLen = Math.Min(rectangles[i][0], rectangles[i][1]);
            arr[i] = newLen;
            maxLen = Math.Max(maxLen, newLen);
        }

        var cnt = 0;
        foreach (var item in arr)
        {
            if (item == maxLen)
                cnt++;
        }

        return cnt;
    }
}
```

**Python**
```
class Solution:
    def countGoodRectangles(self, rectangles: List[List[int]]) -> int:
        n, arr, max_len = len(rectangles), [], 0
        for i in range(n):
            new_len = min(rectangles[i][0], rectangles[i][1])
            arr.append(new_len)
            max_len = max(max_len, new_len)
        
        cnt = 0
        for item in arr:
            if item == max_len:
                cnt += 1
        return cnt
        
```

- Recommendation by one time enumerating
```
class Solution:
    def countGoodRectangles(self, rectangles: List[List[int]]) -> int:
        res, max_len = 0, 0
        for l, w in rectangles:
            k = min(l, w)
            if k == max_len:
                res += 1
            elif k > max_len:
                res = 1
                max_len = k
        return res
        
```
