- Solution by enumeration
```
public class Solution {
    public int NearestValidPoint(int x, int y, int[][] points) {
        var minManhattanDistance = 0;
        var minIdx = -1;
        for (var i = 0; i < points.Length; i++)
        {
            if (points[i][0] == x || points[i][1] == y)
            {
                var currentDistance = Math.Abs(x - points[i][0]) + Math.Abs(y - points[i][1]);
                if (minManhattanDistance == 0 && minIdx == -1)
                {
                    minManhattanDistance = currentDistance;
                    minIdx = i;
                }
                else if (currentDistance < minManhattanDistance)
                {
                    minManhattanDistance = currentDistance;
                    minIdx = i;
                }
            }
        }
        return minIdx;
    }
}

class Solution:
    def nearestValidPoint(self, x: int, y: int, points: List[List[int]]) -> int:
        min_manhattan_distance, min_idx = 0, -1
        for i, point in enumerate(points):
            if point[0] == x or point[1] == y:
                cur_distance = abs(x - point[0]) + abs(y - point[1])
                if min_manhattan_distance == 0 and min_idx == -1:
                    min_manhattan_distance = cur_distance
                    min_idx = i
                elif cur_distance < min_manhattan_distance:
                    min_manhattan_distance = cur_distance
                    min_idx = i
        return min_idx

- Recommendation by enumerating all the points
```
class Solution:
    def nearestValidPoint(self, x: int, y: int, points: List[List[int]]) -> int:
        n = len(points)
        best, best_idx = float("inf"), -1
        for i, (px, py) in enumerate(points):
            if x == px:
                if (dist := abs(y - py)) < best:
                    best = dist
                    best_idx = i
            elif y == py:
                if (dist := abs(x - px)) < best:
                    best = dist
                    best_idx = i
        return best_idx
