**https://leetcode.cn/problems/time-needed-to-buy-tickets/description/**

- Solution by copilot with queue
```
public class Solution {
    public int TimeRequiredToBuy(int[] tickets, int k) {
        var cnt = 0;
        var queue = new Queue<(int ticket, int index)>();

        for (var i = 0; i < tickets.Length; i++)
            queue.Enqueue((tickets[i], i));

        while (queue.Count > 0)
        {
            var (t, i) = queue.Dequeue();
            cnt++;
            if (t - 1 > 0)
                queue.Enqueue((t - 1, i));
            else if (i == k)
                return cnt;
        }

        return cnt;
    }
}

**Python**
```

from collections import deque


class Solution:
    def timeRequiredToBuy(self, tickets: List[int], k: int) -> int:
        cnt = 0
        queue = deque((t, i) for i, t in enumerate(tickets))

        while queue:
            t, i = queue.popleft()
            cnt += 1
            if t - 1 > 0:
                queue.append((t - 1, i))
            elif i == k:
                return cnt

        return cnt
        
```

- Recommendation by counting for everyone
```
class Solution:
    def timeRequiredToBuy(self, tickets: List[int], k: int) -> int:
        n = len(tickets)
        res = 0
        for i in range(n):
            if i <= k:
                res += min(tickets[i], tickets[k])
            else:
                res += min(tickets[i], tickets[k] - 1)
        return res
