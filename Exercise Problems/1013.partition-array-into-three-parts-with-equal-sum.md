public class Solution {
    public bool CanThreePartsEqualSum(int[] arr) {
        var s = 0;
        foreach (var num in arr)
        {
            s += num;
        }
        if (s % 3 != 0) return false;

        var target = s / 3;
        var n = arr.Length;
        var i = 0;
        var cur = 0;
        while (i < n)
        {
            cur += arr[i];
            if (cur == target) break;
            i++;
        }
        if (cur != target) return false;
        var j = i + 1;
        while (j + 1 < n)
        {
            cur += arr[j];
            if (cur == target * 2) return true;
            j++;
        }

        return false;
    }
}

class Solution:
    def canThreePartsEqualSum(self, arr: List[int]) -> bool:
        s = sum(arr)
        if s % 3 != 0:
            return False
        target = s / 3
        n, i, cur = len(arr), 0, 0
        while i < n:
            cur += arr[i]
            if cur == target:
                break;
            i += 1
        if cur != target:
            return False
        j = i + 1
        while j + 1 < n:
            cur += arr[j]
            if cur == target * 2:
                return True
            j += 1
        return False
