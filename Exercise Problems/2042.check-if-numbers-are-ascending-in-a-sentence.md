- Solution by simulation
```
public class Solution {
    public bool AreNumbersAscending(string s) {
        var tokenArray = s.Split(' ', StringSplitOptions.RemoveEmptyEntries);
        var prevNum = -1;
        foreach (var token in tokenArray)
        {
            if (int.TryParse(token, out var n))
            {
                if (prevNum >= n)
                    return false;
                prevNum = n;
            }
        }
        return true;
    }
}

- Solution by simulation with O(1) space optimization
```
public class Solution {
    public bool AreNumbersAscending(string s) {
        var prevNum = -1;
        var temp = string.Empty;
        foreach (var ch in s)
        {
            if (char.IsDigit(ch))
                temp += ch;
            else if (ch == ' ' && temp.Length > 0)
            {
                var n = int.Parse(temp);
                if (prevNum >= n)
                    return false;
                prevNum = n;
                temp = string.Empty;
            }
        }
        if (temp.Length > 0)
        {
            var n = int.Parse(temp);
            if (prevNum >= n)
                return false;
        }
        
        return true;
    }
}
```

class Solution:

            elif ch == ' ' and len(temp) > 0:
                n = int(temp)

                    return False
                prev_num = n
                temp = ""

            n = int(temp)

- Recommendation by direct enumerating
```
class Solution:
    def areNumbersAscending(self, s: str) -> bool:
        prev = i = 0
        while i < len(s):
            if s[i].isdigit():
                cur = 0
                while i < len(s) and s[i].isdigit():
                    cur = cur * 10 + int(s[i])
                    i += 1
                if cur <= prev:
                    return False
                prev = cur
            else:
                i += 1
        return True
