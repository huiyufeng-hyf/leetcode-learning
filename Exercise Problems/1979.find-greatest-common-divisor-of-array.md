public class Solution {
    public int FindGCD(int[] nums) {
        Array.Sort(nums);
        var n = nums.Length;
        var min = nums[0];
        var max = nums[n - 1];
        return GCD(min, max);
    }

    private int GCD(int a, int b)
    {
        while (b != 0)
        {
            var temp = b;
            b = a % b;
            a = temp;
        }
        return a;
    }
}

class Solution:
    def findGCD(self, nums: List[int]) -> int:
        nums.sort()
        n = len(nums)
        min_d, max_d = nums[0], nums[n - 1]
        return self.GCD(min_d, max_d)

    def GCD(self, a: int, b: int) -> int:
        while (b != 0):
            temp = b
            b = a % b
            a = temp
        return a

import math

class Solution:
    def findGCD(self, nums: List[int]) -> int:
        mx, mn = max(nums), min(nums)
        return math.gcd(mx, mn)
