- Solution 1: (Recommendation 1)
```
public class Solution {
    public int Tribonacci(int n) {
        if (n == 0) return 0;
        if (n <= 2) return 1;

        int p = 0, q = 0, r = 1, s = 1;
        for (var i = 3; i <= n; i++)
        {
            p = q;
            q = r;
            r = s;
            s = p + q + r;
        }
        
        return s;
    }
}

- Solution 2: (Recommendation 2)
```
public class Solution {
    public int Tribonacci(int n) {
        if (n == 0) return 0;
        if (n <= 2) return 1;

        int[,] q = {{1, 1, 1}, {1, 0, 0}, {0, 1, 0}};
        int[,] res = Pow(q, n);
        return res[0, 2];
    }

    private int[,] Pow(int[,] a, int n)
    {
        int[,] ret = {{1, 0, 0}, {0, 1, 0}, {0, 0, 1}};
        while (n > 0)
        {
            if ((n & 1) == 1)
                ret = Multiply(ret, a);
            n >>=1;
            a = Multiply(a, a);
        }

        return ret;
    }

    private int[,] Multiply(int[,] a, int[,] b)
    {
        int[,] c = new int[3, 3];
        for (var i = 0; i < 3; i++)
        {
            for (var j = 0; j < 3; j++)
            {
                c[i, j] = a[i, 0] * b[0, j] + a[i, 1] * b[1, j] + a[i, 2] * b[2, j];
            }
        }
        return c;
    }
}
