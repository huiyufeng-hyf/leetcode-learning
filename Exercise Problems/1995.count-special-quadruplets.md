**https://leetcode.cn/problems/count-special-quadruplets/description/**

- Solution by simulation
```
public class Solution {
    public int CountQuadruplets(int[] nums) {
        var n = nums.Length;
        var cnt = 0;
        for (var i = 0; i < n - 3; i++)
        {
            for (var j = i + 1; j < n - 2; j++)
            {
                for (var k = j + 1; k < n - 1; k++)
                {
                    for (var l = k + 1; l < n; l++)
                    {
                        if (nums[i] + nums[j] + nums[k] == nums[l])
                            cnt++;
                    }
                }
            }
        }
        return cnt;
    }
}
```

class Solution:
    def countQuadruplets(self, nums: List[int]) -> int:
        n, cnt = len(nums), 0
        for i in range(n - 3):
            for j in range(i + 1, n - 2):
                for k in range(j + 1, n - 1):
                    for l in range(k + 1, n):
                        if nums[i] + nums[j] + nums[k] == nums[l]:
                            cnt += 1
        return cnt

class Solution:
    def countQuadruplets(self, nums: List[int]) -> int:
        n = len(nums)
        ans = 0
        for a in range(n):
            for b in range(a + 1, n):
                for c in range(b + 1, n):
                    for d in range(c + 1, n):
                        if nums[a] + nums[b] + nums[c] == nums[d]:
                            ans += 1
        return ans

- Recommendation 2 by hashset for nums[d]
```
class Solution:
    def countQuadruplets(self, nums: List[int]) -> int:
        n = len(nums)
        ans = 0
        cnt = Counter()
        for c in range(n - 2, 1, -1):
            cnt[nums[c + 1]] += 1
            for a in range(c):
                for b in range(a + 1, c):
                    if (total := nums[a] + nums[b] + nums[c]) in cnt:
                        ans += cnt[total]
        return ans
        
```

- Recommendation 3 by hashset for nums[d] - nums[c]
```
class Solution:
    def countQuadruplets(self, nums: List[int]) -> int:
        n = len(nums)
        ans = 0
        cnt = Counter()
        for b in range(n - 3, 0, -1):
            for d in range(b + 2, n):
                cnt[nums[d] - nums[b + 1]] += 1
            for a in range(b):
                if (total := nums[a] + nums[b]) in cnt:
                    ans += cnt[total]
        return ans
        
```
