public class Solution {
    public IList<bool> PrefixesDivBy5(int[] nums) {
        var result = new List<bool>();
        var prefix = 0;
        var length = nums.Length;
        for (var i = 0; i < length; i++)
        {
            prefix = ((prefix << 1) + nums[i]) % 5;
            result.Add(prefix == 0);
        }

        return result;
    }
}
