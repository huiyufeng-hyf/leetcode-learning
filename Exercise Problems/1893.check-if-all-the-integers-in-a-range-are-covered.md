**https://leetcode.cn/problems/check-if-all-the-integers-in-a-range-are-covered/description/**

- Solution by enumeration
```
public class Solution {
    public bool IsCovered(int[][] ranges, int left, int right) {
        for (var x = left; x <= right; x++)
        {
            var flag = false;
            foreach (var r in ranges)
            {
                if (x >= r[0] && x <= r[1])
                {
                    flag = true;
                    continue;
                }
            }
            if (!flag)
                return false;
        }
        return true;
    }
}
```

class Solution:
    def isCovered(self, ranges: List[List[int]], left: int, right: int) -> bool:
        for x in range(left, right+1):
            flag = False
            for r in ranges:
                if x >= r[0] and x <= r[1]:
                    flag = True
                    continue
            if not flag:
                return False
        return True

- Recommendation by maintaining diff array
```
class Solution:
    def isCovered(self, ranges: List[List[int]], left: int, right: int) -> bool:
        diff = [0] * 52
        for l, r in ranges:
            diff[l] += 1
            diff[r+1] -= 1
        curr = 0
        for i in range(1, 51):
            curr += diff[i]
            if left <= i <= right and curr <= 0:
                return False
        return True
        
```
