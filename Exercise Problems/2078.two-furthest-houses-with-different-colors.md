**https://leetcode.cn/problems/two-furthest-houses-with-different-colors/description/**

- Solution by enumeration
```
public class Solution {
    public int MaxDistance(int[] colors) {
        var maxDist = 0;
        var n = colors.Length;
        for (var i = 0; i < n; i++)
        {
            for (var j = i + 1; j < n; j++)
            {
                if (colors[i] == colors[j])
                    continue;
                maxDist = Math.Max(maxDist, Math.Abs(i - j));
            }
        }
        return maxDist;
    }
}
```

```
class Solution:
    def maxDistance(self, colors: List[int]) -> int:
        max_dist, n = 0, len(colors)
        for i in range(n):
            for j in range(i + 1, n):
                if colors[i] == colors[j]:
                    continue
                max_dist = max(max_dist, abs(i - j))
        return max_dist
```

- Recommendation by enumeration
```
class Solution:
    def maxDistance(self, colors: List[int]) -> int:
        n = len(colors)
        res = 0
        for i in range(n):
            for j in range(i + 1, n):
                if colors[i] != colors[j]:
                    res = max(res, j - i)
        return res
```
