- Solution by simulation
```
public class Solution {
    public int MinTimeToType(string word) {
        var cnt = 0;
        var n = word.Length;
        var temp = 'a';
        foreach (var ch in word)
        {
            var d = Math.Abs(ch - temp);
            temp = ch;
            cnt += Math.Min(d, 26 -d);
        }
        cnt += n;
        return cnt;
    }
}
```

**Python**
```
class Solution:
    def minTimeToType(self, word: str) -> int:
        cnt, n = 0, len(word)
        temp = 'a'
        for ch in word:
            d = abs(ord(ch) - ord(temp))
            temp = ch
            cnt += min(d, 26 - d)
        cnt += n
        return cnt
        
```

```
class Solution:
    def minTimeToType(self, word: str) -> int:
        prev, res = 0, 0
        for ch in word:
            curr = ord(ch) - ord('a')
            res += 1 + min(abs(curr - prev), 26 - abs(curr - prev))
            prev = curr
        return res
        
```
