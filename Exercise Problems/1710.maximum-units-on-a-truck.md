- Solution by dictionary
```
public class Solution {
    public int MaximumUnits(int[][] boxTypes, int truckSize) {
        var dict = new Dictionary<int, int>();
        foreach (var box in boxTypes)
        {
            if (!dict.ContainsKey(box[1]))
                dict.Add(box[1], box[0]);
            else
                dict[box[1]] += box[0];
        }

        var sortedDict = dict.OrderByDescending(i => i.Key);
        var cnt = 0;
        var maxUnits = 0;
        foreach (var item in sortedDict)
        {
            if (cnt + item.Value <= truckSize)
            {
                maxUnits += item.Key * item.Value;
                cnt += item.Value;
            }
            else
            {
                maxUnits += (truckSize - cnt) * item.Key;
                break;
            }
        }

        return maxUnits;
    }
}
```

**Python**
```
class Solution:
    def maximumUnits(self, boxTypes: List[List[int]], truckSize: int) -> int:
        box_dict = {}
        for box in boxTypes:
            units_per_box = box[1]
            num_boxes = box[0]
            if units_per_box not in box_dict:
                box_dict[units_per_box] = num_boxes
            else:
                box_dict[units_per_box] += num_boxes
            
        sorted_items = sorted(box_dict.items(), key=lambda x: x[0], reverse=True)
        cnt, max_units = 0, 0
        for units_per_box, num_boxes in sorted_items:
            if cnt + num_boxes <= truckSize:
                max_units += units_per_box * num_boxes
                cnt += num_boxes
            else:
                max_units += (truckSize - cnt) * units_per_box
                break
        
        return max_units
        
```

- Recommendation by sorting and greedy
```
class Solution:
    def maximumUnits(self, boxTypes: List[List[int]], truckSize: int) -> int:
        boxTypes.sort(key=lambda x: x[1], reverse=True)
        res = 0
        for num_boxes, units_per_box in boxTypes:
            if num_boxes >= truckSize:
                res += truckSize * units_per_box
                break
            res += num_boxes * units_per_box
            truckSize -= num_boxes
        return res
        
```
