- Solution by simulation
```
public class Solution {
    public string CapitalizeTitle(string title) {
        var strArr = title.Split(' ', StringSplitOptions.RemoveEmptyEntries);
        var sb = new StringBuilder();
        for (var i = 0; i < strArr.Length; i++)
        {
            if (strArr[i].Length <= 2)
                sb.Append(strArr[i].ToLower());
            else
                sb.Append(char.ToUpper(strArr[i][0])).Append(strArr[i].Substring(1).ToLower());
            if (i != strArr.Length - 1)
                sb.Append(' ');
        }
        return sb.ToString();
    }
}

**Python**
```
class Solution:
    def capitalizeTitle(self, title: str) -> str:
        str_arr = title.split(' ')
        new_arr = []
        for i, item in enumerate(str_arr):
            if len(str_arr[i]) <= 2:
                new_arr.append(str_arr[i].lower())
            else:
                new_arr.append(str_arr[i][0].upper() + str_arr[i][1:].lower())
            if i != len(str_arr) - 1:
                new_arr.append(' ')
        return "".join(new_arr)

class Solution:
    def capitalizeTitle(self, title: str) -> str:
        res = []
        for word in title.split():
            if len(word) <= 2:
                res.append(word.lower())
            else:
                res.append(word[0].upper() + word[1:].lower())
        return ' '.join(res)
