public class Solution {
    public int NumRookCaptures(char[][] board) {
        var cnt = 0;
        var st = 0;
        var ed = 0;
        var dx = new int[] {0, 1, 0, -1};
        var dy = new int[] {1, 0, -1, 0};
        for (var i = 0; i < 8; i++)
        {
            for (var j = 0; j < 8; j++)
            {
                if (board[i][j] == 'R')
                {
                    st = i; 
                    ed = j;
                    break;
                }
            }
        }

        for (var i = 0; i < 4; i++)
        {
            for  (var step = 0; ; step++)
            {
                var tx = st + step * dx[i];
                var ty = ed + step * dy[i];
                if (tx < 0 || tx >= 8 || ty < 0 || ty >= 8 || board[tx][ty] == 'B')
                    break;

                if (board[tx][ty] == 'p')
                {
                    cnt++;
                    break;
                }
            }
        }

        return cnt;
    }
}
