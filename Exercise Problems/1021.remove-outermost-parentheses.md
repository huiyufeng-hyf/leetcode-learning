- Solution 1: (Recommendation 1)
```
public class Solution {
    public string RemoveOuterParentheses(string s) {
        var sb = new StringBuilder();
        var stack = new Stack<char>();
        foreach (var c in s)
        {
            if (c == ')')
                stack.Pop();
            if (stack.Count > 0)
                sb.Append(c);
            if (c == '(')
                stack.Push(c);
        }

        return sb.ToString();
    }
}

- Solution 2: (Recommendation 2)
```
public class Solution {
    public string RemoveOuterParentheses(string s) {
        var sb = new StringBuilder();
        var level = 0;
        foreach (var c in s)
        {
            if (c == ')')
                level--;
            if (level > 0)
                sb.Append(c);
            if (c == '(')
                level++;
        }

        return sb.ToString();
    }
}
