public class Solution {
    public string MaximumTime(string time) {
        var t = time.ToCharArray(); // format hh:mm

        // Hour tens
        if (t[0] == '?')
            t[0] = (t[1] != '?' && t[1] > '3') ? '1' : '2';

        // Hour units
        if (t[1] == '?')
            t[1] = (t[0] == '2') ? '3' : '9';

        // Minute tens
        if (t[3] == '?')
            t[3] = '5';

        // Minute units
        if (t[4] == '?')
            t[4] = '9';

        return new string(t);
    }
}

class Solution:
    def maximumTime(self, time: str) -> str:
        t = []
        for ch in time:
            t.append(ch)
        if t[0] == '?':
            t[0] = '1' if t[1] != '?' and t[1] > '3' else '2'
        if t[1] == '?':
            t[1] = '3' if t[0] == '2' else '9'
        if t[3] == '?':
            t[3] = '5'
        if t[4] == '?':
            t[4] = '9'
        
        return "".join(t)
