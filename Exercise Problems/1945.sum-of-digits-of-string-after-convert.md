```
public class Solution {
    public int GetLucky(string s, int k) {
        var sum = 0;
        var cnt = 0;
        foreach (var ch in s)
        {
            var d = ch - 'a' + 1;
            if (d > 9)
                sum += d / 10 + d % 10;
            else
                sum += d;
        }
        cnt++;
        
        if (cnt == k || sum < 10)
            return sum; 

        while (cnt < k)
        {
            var temp = 0;
            while (sum > 0)
            {
                temp += sum % 10;
                sum /= 10;
            }
            if (temp < 10)
                return temp;
            sum = temp;
            cnt++;
        }
        return sum;
    }
}
```

```
class Solution:
    def getLucky(self, s: str, k: int) -> int:
        sum = cnt = 0
        for ch in s:
            d = ord(ch) - ord('a') + 1
            if d > 9:
                sum += d // 10 + d % 10
            else:
                sum += d
        cnt += 1

        if cnt == k or sum < 10:
            return sum
        
        while cnt < k:
            temp = 0
            while sum > 0:
                temp += sum % 10
                sum //= 10
            if temp < 10:
                return temp
            sum = temp
            cnt += 1
        return sum
```

```
class Solution:
    def getLucky(self, s: str, k: int) -> int:
        digits = "".join(str(ord(ch) - ord('a') + 1) for ch in s)

        for i in range(k):
            if len(digits) == 1:
                break
            total = sum(int(ch) for ch in digits)
            digits = str(total)
        return int(digits)
```
