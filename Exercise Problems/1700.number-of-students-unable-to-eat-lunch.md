- Solution by using queue
```
public class Solution {
    public int CountStudents(int[] students, int[] sandwiches) {
        var queue = new Queue<int>();
        foreach (var item in students)
            queue.Enqueue(item);

        var n = queue.Count;
        var cnt = 0;
        while (cnt < n)
        {
            var first = queue.Peek();
            if (first == sandwiches[0])
            {
                queue.Dequeue();
                sandwiches = sandwiches[1..];
                cnt = 0;
                n = queue.Count;
            }
            else
            {
                queue.Dequeue();
                queue.Enqueue(first);
                cnt++;
            }
        }
        
        return cnt;
    }
}
```

**Python**
```
from collections import deque


class Solution:
    def countStudents(self, students: List[int], sandwiches: List[int]) -> int:
        q = deque()
        for item in students:
            q.append(item)
        
        n, cnt = len(q), 0
        while cnt < n:
            first = q.popleft()
            if first == sandwiches[0]:
                sandwiches = sandwiches[1:]
                cnt = 0
                n = len(q)
            else:
                q.append(first)
                cnt += 1
        
        return cnt
        
```

```
class Solution:
    def countStudents(self, students: List[int], sandwiches: List[int]) -> int:
        s1 = sum(students)
        s0 = len(students) - s1
        for x in sandwiches:
            if x == 0 and s0:
                s0 -= 1
            elif x == 1 and s1:
                s1 -= 1
            else:
                break
        return s0 + s1
```
