public class Solution {
    public bool IsValid(string s) {
        var len = s.Length;
        if (len <= 1) return false;
        if (len % 2 != 0) return false;
        if (s[len - 1] == '(' || s[len - 1] == '{' || s[len - 1] == '[') return false;
        if (len > 2 && (s.IndexOf(')') > s.IndexOf('(') + 1 || s.IndexOf('}') > s.IndexOf('{') + 1 || s.IndexOf(']') > s.IndexOf('[') + 1))  
            return false;
        var bracketDict = new Dictionary<char, int>{
            {'(', 0},
            {')', 0},
            {'{', 0},
            {'}', 0},
            {'[', 0},
            {']', 0}
        };
        for (var i = 0; i < len; i++)
        {
            switch (s[i])
            {
                case '(':
                    if (i == 0 && s.IndexOf(')') % 2 == 0) return false;
                    bracketDict['('] += 1;
                    break;
                case ')':
                    if (i == 0) return false;
                    bracketDict[')'] += 1;
                    break;
                case '{':
                    if (i == 0 && s.IndexOf('}') % 2 == 0) return false;
                    bracketDict['{'] += 1;
                    break;
                case '}':
                    if (i == 0) return false;
                    bracketDict['}'] += 1;
                    break;
                case '[':
                    if (i == 0 && s.IndexOf(']') % 2 == 0) return false;
                    bracketDict['['] += 1;
                    break;
                case ']':
                    if (i == 0) return false;
                    bracketDict[']'] += 1;
                    break;
                default:
                    break;
            }
        }

        if (bracketDict['('] != bracketDict[')'] || bracketDict['{'] != bracketDict['}'] || bracketDict['['] != bracketDict[']'])
            return false;

        return true;
    }
}

public class Solution {
    public bool IsValid(string s) {
                var len = s.Length;
        if (len % 2 == 1) return false;

        var pairs = new Dictionary<char, char>{
            {')', '('},
            {']', '['},
            {'}', '{'}
        };

        var stack = new Stack();
        for (var i = 0; i < len; i++)
        {
            var ch = s[i];
            if (pairs.ContainsKey(ch))
            {
                if (stack.Count == 0 || (char)stack.Peek() != pairs[ch])
                    return false;
                stack.Pop();
            }
            else
            {
                stack.Push(ch);
            }
        }
        return stack.Count == 0;
    }
}
