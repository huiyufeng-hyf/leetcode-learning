- Solution 1: (Recommendation 1)
```
public class Solution {
    public IList<int> AddToArrayForm(int[] num, int k) {
        var res = new List<int>();
        var n = num.Length;
        for (var i = n - 1; i >= 0; i--)
        {
            var sum = num[i] + k % 10;
            k /= 10;
            if (sum >= 10)
            {
                k++;
                sum -= 10;
            }
            res.Add(sum);
        }
        for (; k > 0; k /= 10)
        {
            res.Add(k % 10);
        }

        res.Reverse();
        return res;
    }
}

- Solution 2: (Recommendation 2)
```
public class Solution {
    public IList<int> AddToArrayForm(int[] num, int k) {
        var res = new List<int>();
        var n = num.Length;
        for (var i = n - 1; i >= 0 || k > 0; i--, k /= 10)
        {
            if (i >= 0)
                k += num[i];
            res.Add(k % 10);
        }

        res.Reverse();
        return res;
    }
}
