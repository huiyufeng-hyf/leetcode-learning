**https://leetcode.cn/problems/find-first-palindromic-string-in-the-array/description/**

- Solution by enumeration
```
public class Solution {
    public string FirstPalindrome(string[] words) {
        foreach (var word in words)
        {
            if (IsPalindromic(word))
                return word;
        }
        return string.Empty;
    }

    private bool IsPalindromic(string word)
    {
        var n = word.Length;
        if (n == 1) return true;
        var flag = true;
        for (var i = 0; i < n / 2; i++)
        {
            if (word[i] != word[n - 1 - i])
            {
                flag = false;
                break;
            }
        }
        return flag;
    }
}
```

**Python**
```
class Solution:
    def firstPalindrome(self, words: List[str]) -> str:
        def is_palindromic(word: str) -> bool:
            n = len(word)
            if n == 1:
                return True
            flag = True
            for i in range(n // 2):
                if word[i] != word[n - 1 - i]:
                    flag = False
                    break
            return flag

        for word in words:
            if is_palindromic(word):
                return word
        return ""
```

- Recommendation by checking each
```
class Solution:
    def firstPalindrome(self, words: List[str]) -> str:
        def is_palindromic(word: str) -> bool:
            n = len(word)
            l, r = 0, n - 1
            while l < r:
                if word[l] != word[r]:
                    return False
                l += 1
                r -= 1
            return True

        for word in words:
            if is_palindromic(word):
                return word
        return ""
```
