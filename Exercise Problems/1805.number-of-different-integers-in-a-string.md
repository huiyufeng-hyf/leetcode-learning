public class Solution {
    public int NumDifferentIntegers(string word) {
        var uniqueNums = new HashSet<string>();
        var tempStr = new StringBuilder();

        foreach ( var ch in word)
        {
            if (char.IsDigit(ch))
                tempStr.Append(ch);
            else if (tempStr.Length > 0)
            {
                uniqueNums.Add(NormalizeNumber(tempStr.ToString()));
                tempStr.Clear();
            }
        }

        if (tempStr.Length > 0)
            uniqueNums.Add(NormalizeNumber(tempStr.ToString()));

        return uniqueNums.Count;
    }

    private string NormalizeNumber(string num)
    {
        // Strip leading zeros, but keep at least one digit
        var i = 0;
        while (i < num.Length - 1 && num[i] == '0')
            i++;
        return num.Substring(i);
    }
}

class Solution:
    def numDifferentIntegers(self, word: str) -> int:
        unique_nums = set()
        temp_str = ""

        for ch in word:
            if ch.isdigit():
                temp_str += ch
            elif temp_str:
                unique_nums.add(self._normalize_number(temp_str))
                temp_str = ""
        
        if temp_str:
            unique_nums.add(self._normalize_number(temp_str))

        return len(unique_nums)


    def _normalize_number(self, num: str) -> str:
        # Strip leading zeros, but keep at least one digit
        return num.lstrip('0') or '0'

class Solution:
    def numDifferentIntegers(self, word: str) -> int:
        s = set()
        n = len(word)
        p1 = 0
        while True:
            while p1 < n and not word[p1].isdigit():
                p1 += 1
            if p1 == n:
                break
            p2 = p1
            while p2 < n and word[p2].isdigit():
                p2 += 1
            while p2 - p1 > 1 and word[p1] == '0': # remove leading 0
                p1 += 1
            s.add(word[p1:p2])
            p1 = p2
        return len(s)
