- Solution by enumeration
```
public class Solution {
    public bool IsSumEqual(string firstWord, string secondWord, string targetWord) {
        var lenFirstWord = firstWord.Length;
        var lenSecondWord = secondWord.Length;
        var lenTargetWord = targetWord.Length;
        var sbFirst = new StringBuilder();
        var sbSecond = new StringBuilder();
        var sbTarget = new StringBuilder();
        var maxLen = Math.Max(Math.Max(lenFirstWord, lenSecondWord), lenTargetWord);
        for (var i = 0; i < maxLen; i++)
        {
            if (i < lenFirstWord)
                sbFirst.Append(firstWord[i] - 'a');
            if (i < lenSecondWord)
                sbSecond.Append(secondWord[i] - 'a');
            if (i < lenTargetWord)
                sbTarget.Append(targetWord[i] - 'a');
        }
        var intFirst = Convert.ToInt32(sbFirst.ToString());
        var intSecond = Convert.ToInt32(sbSecond.ToString());
        var intTarget = Convert.ToInt32(sbTarget.ToString());
        return (intFirst + intSecond) == intTarget;
    }
}
```

```
class Solution:
    def isSumEqual(self, firstWord: str, secondWord: str, targetWord: str) -> bool:
        len_first, len_second, len_target = len(firstWord), len(secondWord), len(targetWord)
        arr_first, arr_second, arr_target = [''] * len_first, [''] * len_second, [''] * len_target
        max_len = max(max(len_first, len_second), len_target)
        for i in range(max_len):
            if i < len_first:
                arr_first[i] = str(ord(firstWord[i]) - ord('a'))
            if i < len_second:
                arr_second[i] = str(ord(secondWord[i]) - ord('a'))
            if i < len_target:
                arr_target[i] = str(ord(targetWord[i]) - ord('a'))
        int_first = int("".join(arr_first))
        int_second = int("".join(arr_second))
        int_target = int("".join(arr_target))
        return (int_first + int_second) == int_target
```

```
class Solution:
    def isSumEqual(self, firstWord: str, secondWord: str, targetWord: str) -> bool:
        def decode(word: str) -> int:
            res = 0
            for ch in word:
                res *= 10
                res += ord(ch) - ord('a')
            return res
        
        return decode(firstWord) + decode(secondWord) == decode(targetWord)
```
