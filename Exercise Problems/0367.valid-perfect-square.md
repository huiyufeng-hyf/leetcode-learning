    public bool IsPerfectSquare(int num) {
        var x = (int) Math.Sqrt(num);
        return x * x == num;
    }
}

- Solution 2: (Recommendation 2)
```
public class Solution {
    public bool IsPerfectSquare(int num) {
        if (num == 1) return true;
        var left = 0;
        var right = num;
        while (left <= right)
        {
            var mid = (right - left) / 2 + left;
            var square = (long) mid * mid;
            if (square < num)
                left = mid + 1;
            else if (square > num)
                right = mid - 1;
            else
                return true;
        }

        return false;
    }
}
```
