- Solution by simulation
```
public class Solution {
    public int CountBalls(int lowLimit, int highLimit) {
        var dict = new Dictionary<int, int>();
        for (var i = lowLimit; i <= highLimit; i++)
        {
            var sum = GetDigitsSum(i);
            if (dict.ContainsKey(sum))
                dict[sum]++;
            else
                dict[sum] = 1;
        }

        var maxCount = 0;
        foreach (var (key, val) in dict)
            maxCount = Math.Max(maxCount, val);
        return maxCount;
    }

    private int GetDigitsSum(int num)
    {
        var sum = 0;
        while (num > 0)
        {
            if (num / 10 > 0)
            {
                sum += num % 10;
                num /= 10;
            }
            else
            {
                sum += num;
                num = 0;
            }
        }
        return sum;
    }
}
```

**Python**
```
class Solution:
    def countBalls(self, lowLimit: int, highLimit: int) -> int:
        def get_digits_sum(num: int) -> int:
            s = 0
            while num > 0:
                if num // 10 > 0:
                    s += num % 10
                    num //= 10
                else:
                    s += num
                    num = 0
            return s

        d = {}
        for i in range(lowLimit, highLimit + 1):
            s = get_digits_sum(i)
            if s in d:
                d[s] += 1
            else:
                d[s] = 1

        max_count = 0
        for v in d.values():
            max_count = max(max_count, v)
        return max_count
```

- Recommendation by hash table in Python
```
class Solution:
    def countBalls(self, lowLimit: int, highLimit: int) -> int:
        count = Counter(sum(map(int, str(i))) for i in range(lowLimit, highLimit + 1))
        return max(count.values())
```
