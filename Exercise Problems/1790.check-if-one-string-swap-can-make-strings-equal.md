- Solution by simulation
```
public class Solution {
    public bool AreAlmostEqual(string s1, string s2) {
        if (s1 == s2)
            return true;
        
        var indices = new List<int>();
        for (var i = 0; i < s1.Length; i++)
        {
            if (s1[i] != s2[i])
                indices.Add(i);
            if (indices.Count > 2)
                return false;
        }
        if (indices.Count == 2)
        {
            if (s1[indices[0]] == s2[indices[1]] && s1[indices[1]] == s2[indices[0]])
                return true;
        }

        return false;
    }
}

class Solution:
    def areAlmostEqual(self, s1: str, s2: str) -> bool:
        if s1 == s2:
            return True
        
        indices = list()
        for i in range(len(s1)):
            if s1[i] != s2[i]:
                indices.append(i)
            if len(indices) > 2:
                return False
        if len(indices) == 2:
            if s1[indices[0]] == s2[indices[1]] and s1[indices[1]] == s2[indices[0]]:
                return True
        
        return False
```

- Recommendation by counting statistics
```
class Solution:
    def areAlmostEqual(self, s1: str, s2: str) -> bool:
        i = j = -1
        for idx, (x, y) in enumerate(zip(s1, s2)):
            if x != y:
                if i < 0:
                    i = idx
                elif j < 0:
                    j = idx
                else:
                    return False
        return i < 0 or j >= 0 and s1[i] == s2[j] and s1[j] == s2[i]
```
