public class Solution {
    public int[][] ImageSmoother(int[][] img) {
        var m = img.Length;
        var n = img[0].Length;
        var ret = new int[m][];
        for (var i = 0; i < m; i++)
        {
            ret[i] = new int[n];
        }

        for (var i = 0; i < m; i++)
        {
            for (var j = 0; j < n; j++)
            {
                var num = 0; 
                var sum = 0;
                for (var x = i - 1; x <= i + 1; x++)
                {
                    for (var y = j - 1; y <= j + 1; y++)
                    {
                        if (x >= 0 && x < m && y >= 0 && y < n)
                        {
                            num++;
                            sum += img[x][y];
                        }
                    }
                }
                ret[i][j] = sum / num;
            }
        }

        return ret;
    }
}
