**https://leetcode.cn/problems/count-elements-with-strictly-smaller-and-greater-elements/description/**

- Solution by enumeration
```
public class Solution {
    public int CountElements(int[] nums) {
        var ans = 0;
        var n = nums.Length;
        Array.Sort(nums);
        var minCnt = 1;
        var maxCnt = 1;
        for (var i = 1; i < n - 1; i++)
        {
            if (nums[i] == nums[0])
                minCnt += 1;
            else if (nums[i] == nums[n - 1])
                maxCnt += 1;
        }
        ans = n - minCnt - maxCnt;
        if (ans < 0) return 0;
        return ans;
    }
}
```
**Python**
```
class Solution:
    def countElements(self, nums: List[int]) -> int:
        ans, n = 0, len(nums)
        nums.sort()
        min_cnt, max_cnt = 1, 1
        for i in range(1, n - 1):
            if nums[i] == nums[0]:
                min_cnt += 1
            elif nums[i] == nums[n - 1]:
                max_cnt += 1
        ans = n - min_cnt - max_cnt
        if ans < 0:
            return 0
        return ans
```

- Recommendation by counting as requirement
```
class Solution:
    def countElements(self, nums: List[int]) -> int:
        smallest = min(nums)
        largest = max(nums)
        res = 0
        for num in nums:
            if smallest < num < largest:
                res += 1
        return res
```
