**https://leetcode.cn/problems/keep-multiplying-found-values-by-two/description/**

- Solution by enumeration
```
public class Solution {
    public int FindFinalValue(int[] nums, int original) {
        var n = nums.Length;
        while (true)
        {
            var flag = false;
            for (var i = 0; i < n; i++)
            {
                if (nums[i] == original)
                {
                    original *= 2;
                    flag = true;
                    break;
                }
            }
            if (!flag)
                break;
        }
        return original;
    }
}
```
**Python**
```
class Solution:
    def findFinalValue(self, nums: List[int], original: int) -> int:
        n = len(nums)
        while True:
            flag = False
            for i in range(n):
                if nums[i] == original:
                    original *= 2
                    flag = True
                    break
            if not flag:
                break
        return original
```

- Recommendation 1 by sorting
```
class Solution:
    def findFinalValue(self, nums: List[int], original: int) -> int:
        nums.sort()
        for num in nums:
            if num == original:
                original *= 2
        return original
```

- Recommendation 2 by hashset
```
class Solution:
    def findFinalValue(self, nums: List[int], original: int) -> int:
        s = set(nums)
        while original in s:
            original *= 2
        return original
```
